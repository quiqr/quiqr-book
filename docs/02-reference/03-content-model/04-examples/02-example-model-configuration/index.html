                    Example 1: minimal one file blog #

                    Here’s an example model/base-file containing a minimal configuration for a blog website.

                    ./model/base.
                         
                    build:
- config: config.toml
  key: default
collections:
- dataformat: yaml
  extension: md
  fields:
  - key: draft
    title: Draft
    type: boolean
  - key: title
    title: Title
    type: string
  - extensions:
    - jpg
    - png
    - jpeg
    - pdf
    - svg
    fields:
    - key: thumb
      title: Thumb
      type: bundle-image-thumbnail
    key: page_related_images
    path: ""
    title: Page related images
    type: bundle-manager
  - key: mainContent
    title: Main content
    type: markdown
  - key: publishdate
    title: Publishdate
    type: hidden
  - key: tags
    title: Tags
    type: chips
  folder: content/post/
  itemtitle: Post
  key: c__post
  title: Posts
hugover: extended_0.76.5
menu:
- key: singles
  menuItems:
  - key: s__about
  title: Pages
- key: collections
  menuItems:
  - key: c__post
  title: Groups
- key: general
  menuItems:
  - key: config
  title: General

- config: config.toml

singles:

  file: config.toml
